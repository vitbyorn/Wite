        // ================================================================
        //  ЯДРО СИНТЕЗАТОРА

        const Synth = (function() {
            let ctx = null;
            let active = false;

            function init() {
                try {
                    if (!ctx) {
                        ctx = new(window.AudioContext || window.webkitAudioContext)();
                    }
                    if (ctx.state === 'suspended') ctx.resume();
                    active = (ctx.state === 'running');
                    updateUI();
                    return active;
                } catch (e) {
                    active = false;
                    updateUI();
                    return false;
                }
            }

            function updateUI() {
                const dot = document.getElementById('statusDot');
                const text = document.getElementById('statusText');
                if (active) {
                    dot.className = 'dot on';
                    text.textContent = '✅ Готово';
                } else {
                    dot.className = 'dot off';
                    text.textContent = '🔇 Нажми "Активировать"';
                }
            }

            // Базовое воспроизведение
            function play(freq, dur, type, vol) {
                if (!active && !init()) return;
                if (!ctx || ctx.state !== 'running') return;
                try {
                    const osc = ctx.createOscillator();
                    const gain = ctx.createGain();
                    osc.type = type || 'square';
                    osc.frequency.value = freq || 440;
                    gain.gain.value = vol || 0.25;
                    osc.connect(gain);
                    gain.connect(ctx.destination);
                    osc.start();
                    osc.stop(ctx.currentTime + (dur || 0.1));
                } catch (e) { /* тихо */ }
            }

            // === ЗАГРУЗКА ФАЙЛОВ ===
            let audioBuffer = null;
            let fileSource = null;

            function loadFile(file) {
                return new Promise((resolve, reject) => {
                    const reader = new FileReader();
                    reader.onload = async (e) => {
                        try {
                            if (!ctx) init();
                            const arrayBuffer = e.target.result;
                            audioBuffer = await ctx.decodeAudioData(arrayBuffer);
                            resolve(audioBuffer);
                        } catch (err) { reject(err); }
                    };
                    reader.onerror = reject;
                    reader.readAsArrayBuffer(file);
                });
            }

            function playFile(volume, rate, loop) {
                if (!audioBuffer || !ctx) return;
                if (!active) init();
                if (ctx.state !== 'running') return;
                try {
                    if (fileSource) { try { fileSource.stop(); } catch (e) {} }
                    fileSource = ctx.createBufferSource();
                    fileSource.buffer = audioBuffer;
                    fileSource.loop = loop || false;
                    fileSource.playbackRate.value = rate || 1;
                    const gain = ctx.createGain();
                    gain.gain.value = volume || 0.5;
                    fileSource.connect(gain);
                    gain.connect(ctx.destination);
                    fileSource.start();
                } catch (e) { /* тихо */ }
            }

            function stopFile() {
                if (fileSource) { try { fileSource.stop(); } catch (e) {} }
                fileSource = null;
            }

            function getBuffer() { return audioBuffer; }

            // События для активации
            document.addEventListener('click', () => { if (!active) init(); });
            document.addEventListener('touchstart', () => { if (!active) init(); });

            return {
                init,
                play,
                loadFile,
                playFile,
                stopFile,
                getBuffer,
                isActive: () => active,
                updateUI,
                getCtx: () => ctx
            };
        })();

            spin: { freq: 750, dur: 0.18, type: 'sawtooth', vol: 0.2, label: 'Вжжик' },

            chord_major: { freq: 440, dur: 0.3, type: 'sine', vol: 0.25, label: 'Мажор', chord: [0, 4, 7] },
            chord_minor: { freq: 440, dur: 0.3, type: 'sine', vol: 0.25, label: 'Минор', chord: [0, 3, 7] }
        };

        // ================================================================
        let sequence = [];

            voices: 3,

            type: 'square',

        GRID_NOTES.forEach((row, ri) => {
            row.forEach((freq, ci) => {

                btn.className = 'grid-btn';

                btn.dataset.col = ci;

                    <span class="label">${ri+1}·${ci+1}</span>
                `;

                    if (e.shiftKey) {
                        // Shift+клик — добавить в последовательность
                        addToSequence(freq, 0.12, 'square', 0.3);
                        btn.classList.add('active');
                        setTimeout(() => btn.classList.remove('active'), 200);
                    } else {
                        // Обычный клик — играть звук
                        Synth.play(freq, 0.15, 'square', 0.25);
                        btn.classList.add('active');
                        setTimeout(() => btn.classList.remove('active'), 150);
                    }
                });

                // Двойной клик — изменить волну
                btn.addEventListener('dblclick', () => {
                    const types = ['square', 'sawtooth', 'sine', 'triangle'];
                    const current = comboParams.type;
                    const idx = (types.indexOf(current) + 1) % types.length;
                    comboParams.type = types[idx];
                    btn.style.borderColor = `rgba(0,255,200,${0.2 + idx * 0.1})`;
                    Synth.play(freq, 0.08, types[idx], 0.2);
                });

                gridEl.appendChild(btn);
            });
        });

        // ================================================================
        //  UI — ПОСЛЕДОВАТЕЛЬНОСТЬ
        // ================================================================
        const seqStepsEl = document.getElementById('sequenceSteps');

        function renderSequence() {
            seqStepsEl.innerHTML = '';
            if (sequence.length === 0) {
                seqStepsEl.innerHTML = '<span style="color:#444;font-size:11px;">Нет нот · клик по гриду с Shift</span>';
                document.getElementById('seqCount').textContent = '0 нот';
                return;
            }
            sequence.forEach((item, idx) => {
                const step = document.createElement('div');
                step.className = 'seq-step';
                step.textContent = Math.round(item.freq);
                step.title = `${item.freq} Гц · ${item.type}`;
                step.addEventListener('click', () => {
                    Synth.play(item.freq, 0.12, item.type, item.vol);
                });
                const remove = document.createElement('span');
                remove.className = 'remove-step';
                remove.textContent = '✕';
                remove.addEventListener('click', (e) => {
                    e.stopPropagation();
                    sequence.splice(idx, 1);
                    renderSequence();
                });
                step.appendChild(remove);
                seqStepsEl.appendChild(step);
            });
            document.getElementById('seqCount').textContent = `${sequence.length} нот`;
        }
        renderSequence();

        function addToSequence(freq, dur, type, vol) {
            sequence.push({ freq, dur: dur || 0.12, type: type || 'square', vol: vol || 0.3 });
            renderSequence();
        }

        // ================================================================
        //  КОМБИНАТОР
        // ================================================================

        function getComboFreqs() {
            const base = comboParams.freq || 440;
            const interval = parseInt(document.getElementById('comboInterval').value) || 4;
            const voices = parseInt(document.getElementById('comboVoices').value) || 3;
            const freqs = [];
            for (let i = 0; i < voices; i++) {
                const offset = i * interval;
                let f = base + offset;
                if (f < 50) f = 50;
                if (f > 4000) f = 4000;
                freqs.push(f);
            }
            return freqs;
        }

        function playCombo() {
            const mode = document.getElementById('comboMode').value;
            const delay = parseInt(document.getElementById('comboDelayVal').value) || 120;
            const freqs = getComboFreqs();
            const dur = comboParams.dur || 0.12;
            const type = comboParams.type || 'square';
            const vol = comboParams.vol || 0.3;

            if (mode === 'chord') {
                freqs.forEach((f, i) => {
                    const v = vol * (1 - i * 0.08);
                    Synth.play(f, dur, type, Math.max(0.05, v));
                });
            } else if (mode === 'sequence') {
                freqs.forEach((f, i) => {
                    const v = vol * (1 - i * 0.05);
                    setTimeout(() => Synth.play(f, dur * 0.8, type, Math.max(0.05, v)), i * delay);
                });
            } else if (mode === 'arpeggio') {
                freqs.forEach((f, i) => {
                    const v = vol * (1 - i * 0.08);
                    setTimeout(() => Synth.play(f, dur * 0.5, type, Math.max(0.05, v)), i * (delay * 0.3));
                });
            }
        }

        function playSequence() {
            if (sequence.length === 0) return;
            const delay = parseInt(document.getElementById('comboDelayVal').value) || 120;
            sequence.forEach((item, idx) => {
                setTimeout(() => {
                    Synth.play(item.freq, item.dur || 0.12, item.type || 'square', item.vol || 0.3);
                }, idx * delay);
            });
        }

        // ================================================================
        //  ГЕНЕРАЦИЯ КОДА
        // ================================================================

        function generateCode() {
            let code = '// ===== NEON SYNTH — СГЕНЕРИРОВАННЫЙ КОД =====\n\n';

            // Базовые пресеты
            Object.keys(PRESETS).forEach(key => {
                const p = PRESETS[key];
                if (p.spark) {
                    code += `function ${key}() {\n`;
                    code += `  playBeep(${p.freq}, ${p.dur}, '${p.type}', ${p.vol});\n`;
                    code +=
                        `  setTimeout(() => playBeep(${Math.round(p.freq*0.78)}, ${(p.dur*1.2).toFixed(2)}, '${p.type}', ${(p.vol*0.9).toFixed(2)}), 80);\n`;
                    code +=
                        `  setTimeout(() => playBeep(${Math.round(p.freq*0.57)}, ${(p.dur*1.4).toFixed(2)}, '${p.type}', ${(p.vol*0.8).toFixed(2)}), 180);\n`;
                    code += `}\n\n`;
                } else if (p.chord) {
                    code += `function ${key}() {\n`;
                    p.chord.forEach((offset, i) => {
                        const f = p.freq + offset * 10;
                        code +=
                            `  playBeep(${f}, ${p.dur}, '${p.type}', ${(p.vol * (1 - i*0.05)).toFixed(2)});\n`;
                    });
                    code += `}\n\n`;
                } else {
                    code += `function ${key}() { playBeep(${p.freq}, ${p.dur}, '${p.type}', ${p.vol}); }\n`;
                }
            });

            // Последовательность
            if (sequence.length > 0) {
                code += `\n// ===== Последовательность (${sequence.length} нот) =====\n`;
                code += `function playSequence() {\n`;
                const delay = parseInt(document.getElementById('comboDelayVal').value) || 120;
                sequence.forEach((item, idx) => {
                    code +=
                        `  setTimeout(() => playBeep(${item.freq}, ${item.dur||0.12}, '${item.type||'square'}', ${item.vol||0.3}), ${idx*delay});\n`;
                });
                code += `}\n`;
            }

            // Текущий звук
            code += `\n// ===== Текущий звук =====\n`;
            code +=
                `// playBeep(${comboParams.freq}, ${comboParams.dur}, '${comboParams.type}', ${comboParams.vol});\n`;

            // Загруженный файл
            const buffer = Synth.getBuffer();
            if (buffer) {
                const fileInput = document.getElementById('soundFileInput');
                const name = fileInput.files && fileInput.files[0] ? fileInput.files[0].name : 'sound';
                const cleanName = name.replace(/\.[^/.]+$/, '').replace(/[^a-zA-Z0-9_]/g, '_');
                const vol = parseFloat(document.getElementById('fileVol').value) || 0.5;
                const rate = parseFloat(document.getElementById('fileRate').value) || 1;
                const loop = document.getElementById('fileLoop').checked;
                code += `\n// ===== Загруженный звук: ${name} =====\n`;
                code += `// Для использования загрузи звук при старте игры:\n`;
                code += `// const audioCtx = new AudioContext();\n`;
                code += `// let audioBuffer;\n`;
                code += `// fetch('${name}').then(r=>r.arrayBuffer()).then(buf=>audioCtx.decodeAudioData(buf)).then(b=>{ audioBuffer=b; });\n\n`;
                code += `function play${cleanName}() {\n`;
                code += `  if (!audioBuffer) return;\n`;
                code += `  const ctx = new (window.AudioContext || window.webkitAudioContext)();\n`;
                code += `  const source = ctx.createBufferSource();\n`;
                code += `  source.buffer = audioBuffer;\n`;
                code += `  source.loop = ${loop};\n`;
                code += `  source.playbackRate.value = ${rate};\n`;
                code += `  const gain = ctx.createGain();\n`;
                code += `  gain.gain.value = ${vol};\n`;
                code += `  source.connect(gain);\n`;
                code += `  gain.connect(ctx.destination);\n`;
                code += `  source.start();\n`;
                code += `}\n`;
            }

            return code;
        }

        // Загрузка файлов
        const fileInput = document.getElementById('soundFileInput');
        const fileName = document.getElementById('fileName');

            });
        };

        // Инициализация
        Synth.updateUI();

        console.log('🎹 NEON SYNTH STUDIO загружен!');
        console.log('📁 Загружай файлы, создавай комбинации, копируй код в игру.');
    </script>
